Donut Chart

<chart-donut
    colors="['red', 'green', 'blue']"
    labels="['Red', 'Green', 'Blue']"
    values="[25, 45, 30]"
></chart-donut>
const chartElement = document.createElement('chart-donut');
chartElement.colors = ['red', 'green', 'blue'];
chartElement.labels = ['Red', 'Green', 'Blue'];
chartElement.values = [25, 45, 30];

document.body.append(this.chartElement);

Mortgage Calculator Input

<mortgage-calc-input name="price" type="currency">
    <label for="price" slot="label">Price</label>
    <input
        type="text"
        id="price"
        slot="input"
        placeholder="123,456"
        pattern="[0-9$,]+"
        maxlength="9"
    />
</mortgage-calc-input>
const mortgageCalcInput = document.createElement('mortgage-calc-input');

mortgageCalc.append(mortgageCalcInput);
